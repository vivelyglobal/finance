function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet; // Memilih tab berdasarkan instruksi dari web app if (data.target === "transactions") { sheet = ss.getSheetByName("Transaction Log"); sheet.appendRow([data.date, data.description, data.category, data.amount, data.currency, data.type]); } else if (data.target === "commissions") { sheet = ss.getSheetByName("Payroll & Payout Hub"); sheet.appendRow([data.date, data.project, data.manager, data.type, data.client, data.val, data.currency, data.status]); } return ContentService.createTextOutput(JSON.stringify({"status": "success"})) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // Mengizinkan web app membaca data tanpa terhalang CORS function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); // Ambil data Transaksi var txSheet = ss.getSheetByName("Transaction Log"); var txData = txSheet.getDataRange().getValues(); // Ambil data Payroll var paySheet = ss.getSheetByName("Payroll & Payout Hub"); var payData = paySheet.getDataRange().getValues(); var result = { transactions: txData.slice(1), // Potong baris header commissions: payData.slice(1) }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
